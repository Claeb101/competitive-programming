import java.util.*;
import java.io.*;

public class Problem2 {
    static long N;
    static long K;
    static long M;

    public static void main(String[] args) throws Exception {
        //"D:\\CalebArulandu\\USACO\\Jan\\Silver\\src\\loan.in"
        Scanner s = new Scanner(new File("D:\\CalebArulandu\\USACO\\Jan\\Silver\\src\\loan.in"));
        PrintWriter write = new PrintWriter(new FileWriter(new File("loan.out")));
        String[] first = s.nextLine().trim().split(" ");
        N = Long.parseLong(first[0]);
        K = Long.parseLong(first[1]);
        M = Long.parseLong(first[2]);


        int largest = 1;
        int out;
        for (int X = 2; X < N; X++) {
            out = tryX(X);
            if(out == -1){
                break;
            } else {
                largest = out;
            }
        }
        write.println(largest);
        write.close();
    }
    public static int tryX(int X){
        int G = 0;
        int days = 0;
        long formula = 0;
        while (days <= K) {
            if (G >= N) {
                return X;
            }
            G += Math.max((N-G)/X, M);
            days++;
        }

        return -1;
    }
}
